macro(add_sources)
    file(RELATIVE_PATH _relPath "${PROJECT_SOURCE_DIR}" "${CMAKE_CURRENT_SOURCE_DIR}")
    foreach (_src ${ARGN})
        if (_relPath)
            list(APPEND SRCS "${_relPath}/${_src}")
        else ()
            list(APPEND SRCS "${_src}")
        endif ()
    endforeach ()
    if (_relPath)
        set(SRCS ${SRCS} PARENT_SCOPE)
    endif ()
endmacro()

macro(add_external)
    file(RELATIVE_PATH _relPath "${PROJECT_SOURCE_DIR}" "${CMAKE_CURRENT_SOURCE_DIR}")
    foreach(_src ${ARGN})
        list(APPEND EXTERNAL "_${_src}")
    endforeach()
    if(_relPath)
        set(EXTERNAL ${EXTERNAL} PARENT_SCOPE)
    endif()
endmacro()

macro(linker_flags)
    file(RELATIVE_PATH _relPath "${PROJECT_SOURCE_DIR}" "${CMAKE_CURRENT_SOURCE_DIR}")
    foreach(_src ${ARGN})
        list(APPEND EMC_FLAGS "${_src}")
    endforeach()
    if(_relPath)
        set(EMC_FLAGS ${EMC_FLAGS} PARENT_SCOPE)
    endif()
endmacro()
